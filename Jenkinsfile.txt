pipeline {
    agent any

    stages {
        stage('Clone Repo') {
            steps {
                git 'https://github.com/praveendev08/aws ci-cd project'
            }
        }

        stage('Build Docker Image') {
            steps {
                sh 'docker build -t username/aws-cicd:latest .'
            }
        }

        stage('Push to Docker Hub') {
            steps {
                withCredentials([string(credentialsId: 'dockerhub-pass', variable: 'PASS')]) {
                    sh '''
                    echo $PASS | docker login -u username --password-stdin
                    docker push username/aws-cicd:latest
                    '''
                }
            }
        }
    }
}
